\tikzmath{
	% Adapted from http://www.cs.northwestern.edu/academics/courses/110/html/fib_rec.html
	function fibonacci(\n) {
		if \n == 0 then {
			return 0;
		} else {
			return fibonacci2(\n, 0, 1);
		};
	};
	function fibonacci2(\n, \p, \q) {
		if \n == 1 then {
			return \q;
		} else {
			return fibonacci2(\n-1, \q, \p+\q);
		};
	};
	int \f, \i;
	for \i in {0,1,...,20}{
		\f = fibonacci(\i);
		print {\f, };
	};
}
